// UPDATE JAM DIGITAL function updateTime() { const now = new Date(); const days = ['Minggu', 'Senin', 'Selasa', 'Rabu', 'Kamis', 'Jumat', 'Sabtu']; const months = ['Jan', 'Feb', 'Mar', 'Apr', 'Mei', 'Jun', 'Jul', 'Agu', 'Sep', 'Okt', 'Nov', 'Des']; document.getElementById('current-day').innerText = `${days[now.getDay()]}, ${now.getDate()} ${months[now.getMonth()]}`; document.getElementById('current-time').innerText = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}:${String(now.getSeconds()).padStart(2, '0')} WIB`; } setInterval(updateTime, 1000); updateTime(); // FUNGSI DETEKSI LOKASI REAL-TIME function getLocation() { const locationElement = document.getElementById('user-location'); if (navigator.geolocation) { navigator.geolocation.getCurrentPosition( // Jika Berhasil async (position) => { const lat = position.coords.latitude; locationElement.innerHTML = ` ${city}, ${country}`; } catch (error) { locationElement.innerHTML = ` Lokasi Aktif`; } }, // Jika Gagal (User menolak izin) () => { locationElement.innerHTML = ` Lokasi Dimatikan`;
